# Police Garage with target

Add this to qb-target config.lua.  
add this under Config.BoxZones = {




["policegarage"] = {

        models = {

            'ig_trafficwarden',

        },

        options = {

            {

                type = "client",

                event = "garage:menu",

                icon = "fas fa-car",

                label = "Police Garage",

                job = "police",

            },

        },

        distance = 3.5,

    },
